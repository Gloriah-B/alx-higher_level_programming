import doctest

def test_say_my_name():
    """
    Test cases for say_my_name function.
    """

    # Valid cases
    assert say_my_name("Bob") == "My name is Bob"
    assert say_my_name("John", "Doe") == "My name is John Doe"
    assert say_my_name("Alice") == "My name is Alice"

    # Invalid cases
    try:
        say_my_name(123)
    except TypeError as e:
        assert str(e) == "first_name must be a string"

    try:
        say_my_name("Bob", 456)
    except TypeError as e:
        assert str(e) == "last_name must be a string"

if __name__ == "__main__":
    test_say_my_name()
    doctest.testmod()
